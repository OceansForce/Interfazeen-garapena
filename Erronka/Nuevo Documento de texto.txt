CREATE TABLE IF NOT EXISTS `Mahiak` (
	`id` INTEGER NOT NULL AUTO_INCREMENT UNIQUE,
	`mahia` INTEGER NOT NULL,
	`erreserbatua` BOOLEAN NOT NULL,
	`data` DATE NOT NULL,
	`mota` TEXT(65535) NOT NULL,
	`erabiltzailea` VARCHAR(100),
	PRIMARY KEY(`id`)
);


CREATE TABLE IF NOT EXISTS `Erabiltzaileak` (
    `Izena` VARCHAR(100) NOT NULL UNIQUE,
    `Pazahitza` TEXT(65535) NOT NULL,
    `Mota` TEXT(65535) NOT NULL,
    PRIMARY KEY (`Izena`)
);


CREATE TABLE IF NOT EXISTS `Biltegia` (
	`id` INTEGER NOT NULL,
	`Izena` TEXT(65535) NOT NULL,
	`Kantitatea` INTEGER NOT NULL,
	`Prezioa` FLOAT,
	PRIMARY KEY(`id`)
);
